            ```powershell

            REM powershell Get-Process | where {$_.processname -eq 'miniserve-v0.19.5-x86_64-pc-windows-msvc'} | Stop-Process -Force

            if %errorlevel% == 0 (
              REM 存在
              REM powershell kill -processname miniserve-v0.19.5-x86_64-pc-windows-msvc
            ```
